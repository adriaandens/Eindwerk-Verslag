int countFiles(){
	int count = 0;
	DIR *d;
	struct dirent *dir;
	d = opendir(".");

	if (d)
	{
		while ((dir = readdir(d)) != NULL)
		{
			if(strcmp(dir->d_name, ".") != 0 && strcmp(dir->d_name, "..") != 0){
				count++;
			}
		}

		closedir(d);
	}

	return count;

}
