f1 = open( "hex_dump", "r" )

        tekst = tekst + ""+ line.rstrip('\n')
f1.close()

n = 2
array = []
array = [tekst[i:i+n] for i in range(0, len(tekst), n)]

# change 4947 to 4749
n = 2

for i in range(0, len(array), n):
# [0] in [1] steken & omgekeerd
